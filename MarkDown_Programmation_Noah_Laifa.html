get_chess_com_player <- function(username) {
  base_url <- paste0("https://api.chess.com/pub/player/", username)
  response <- GET(base_url)
  if (status_code(response) == 200) {
    return(fromJSON(rawToChar(response$content)))
  }
  return(NULL)
}
